Script to Hurt/Heal Player

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class HealPlayer : MonoBehaviour
{

public int amount;
//collision detector function
private void OnCollisionEnter(Collision hit)
{

if (hit.gameObject.tag == "Player") //if a player is colliding
{

var healthComponent = hit.gameObject.GetComponent<Health>();
//GetComponent<Health>(); //gets the health of the player that collided
if (healthComponent != null) //if the player DOES have a health component {

healthComponent.HealDamage(amount); //runs take damage function

}

}

}

}
