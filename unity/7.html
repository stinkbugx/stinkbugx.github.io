Script for Enemy Chasing

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class chasePlayer : MonoBehaviour
{

//set the values in the inspector
public Transform target; //drag and stop player object in the inspector
public float range;
public float speed;

Rigidbody rb; //variable we will use to access rigid body

void Start()
{

rb = GetComponent<Rigidbody>(); //accesses the rigid body
rb.freezeRotation = true; //freezes the rigid body rotation

}

public void FixedUpdate()
{

//get the distance between the player and enemy (this object)
float dist = Vector3.Distance(target.position, transform.position);
//check if it is within the range you set
if (dist <= range)
{

//move to target(player)
transform.position = Vector3.MoveTowards(transform.position, target.transform.position, speed/20);

}
//else, if it is not in rage, it will not follow player

}

}
