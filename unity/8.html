Script to Shoot

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class shoot : MonoBehaviour
{

public float projectileSpeed;
public projectile projectilePrefab;
public float coolDown;
private bool readyToShoot;
public Transform orientation;

//or get rid of these two if you want mouse click
[Header("Keybinds")]
public KeyCode shootKey = KeyCode.Q;

private void Start()
{

readyToShoot = true;

}

private void Update()
{

//or Input.GetMouseButtonDown(0)
if (Input.GetKey(shootKey) && readyToShoot)
{

readyToShoot = false;
var position = transform.position + orientation.forward;
var rotation = orientation.rotation;
var projectile = Instantiate(projectilePrefab, position, rotation);
projectile.Fire(projectileSpeed, orientation.forward);

Invoke(nameof(ResetShoot), coolDown);

}

}

private void ResetShoot()
{

readyToShoot = true;

}

}
