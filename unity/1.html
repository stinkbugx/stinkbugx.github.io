Script to Rotate Camera

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class SCRIPT NAME HERE : MonoBehaviour
{

//changeable sensitivity variables for x and y
public float sensX;
public float sensY;

//transform variable to connect to the object's tranformation properties
public Transform orientation;

//rotation variables for x and y
float rotationX;
float rotationY;

void Start()
{

Cursor.lockState = CursorLockMode.Locked; //locks cursor to the middle of the screen
Cursor.visible = false; //makes the cursor invisible

}

private void Update()
{

//getting mouse inputs for x and y
float mouseX = Input.GetAxisRaw("Mouse X") * Time.deltaTime * sensX;
float mouseY = Input.GetAxisRaw("Mouse Y") * Time.deltaTime * sensY;

//math for rotation respective to mouse input, convoluted but correct
rotationY += mouseX; //adds mouseX variable to rotationY
rotationX -= mouseY; //subtracts mouseY from rotationX

rotationX = Mathf.Clamp(rotationX, -90f, 90f); //restricts the x rotation from going too far

//rotating the character and camera
transform.rotation = Quaternion.Euler(rotationX, rotationY, 0); //0 for the z value
orientation.rotation = Quaternion.Euler(0, rotationY, 0); //0 for x and z, only need to change y for camera since it follows the player

}

}
